# Build a Todo List

This guide walks through building a working todo list from scratch. Each todo item is its own entity — toggling, deleting, and rendering all operate on individual entities rather than a shared array. Along the way you'll use:

- DOM rendering (`DOMElement`, `TextContent`, `Classes`, `Style`)
- inline click handlers (`Clickable` with `onClick`)
- entity-per-item state with structured data (`Status`)
- parent/child entity relationships
- reactive systems for lifecycle and updates

## 1. Define Your Components

Each todo item will be an entity with `TodoText` and `Status` components:

<div data-code-tabs>

```tsx
import { defineComponent, defineMarker } from "@ecs-test/ecs";

// TodoText holds the task text.
// Status stores completion state as explicit data.
const TodoText = defineComponent<{ value: string }>("TodoText");
const Status = defineComponent<{ completed: boolean }>("Status");

// Markers to tag key UI entities so systems can find them
const TodoInput = defineMarker("TodoInput");
const TodoListContainer = defineMarker("TodoListContainer");
```

Notice there's no `Todos` array. Each todo is its own entity — `TodoText` holds the text, and `Status.completed` holds completion state.

## 2. Build the Static UI Shell

The app has a heading, an input row, and an empty container where todo entities will appear as children:

<div data-code-tabs>

```tsx
import { Entity, materialize } from "@ecs-test/ecs";
import { DOMElement, TextContent, Classes, Style, Clickable, getDOMElement } from "@ecs-test/dom";

function addTodoFromInput(world) {
  const inputEntity = world.query(TodoInput)[0];
  if (inputEntity == null) return;
  const inputEl = getDOMElement(world, inputEntity) as HTMLInputElement;
  if (!inputEl) return;

  const text = inputEl.value.trim();
  if (!text) return;
  inputEl.value = "";

  const listEntity = world.query(TodoListContainer)[0];
  if (listEntity == null) return;
  world.createEntity(listEntity, [TodoText({ value: text }), Status({ completed: false })]);
}

const ui = (
  <Entity>
    <DOMElement tag="section" />
    <Classes list={["live-card"]} />

    <Entity>
      <DOMElement tag="h3" />
      <TextContent value="Todo List" />
    </Entity>

    {/* Input row: text field + add button */}
    <Entity>
      <DOMElement tag="div" />
      <Style display="flex" gap="8px" marginBottom="12px" />

      <Entity>
        <DOMElement tag="input" />
        <TodoInput />
        <Style flex="1" padding="6px 10px" border="1px solid #d1d5db"
               borderRadius="4px" font="inherit" />
      </Entity>

      <Entity>
        <DOMElement tag="button" />
        <TextContent value="Add" />
        <Clickable onClick={(world) => addTodoFromInput(world)} />
      </Entity>
    </Entity>

    {/* Empty container — todo entities will be added here as children */}
    <Entity>
      <DOMElement tag="div" />
      <TodoListContainer />
    </Entity>
  </Entity>
);

materialize(world, ui);
```

## 3. Render Todo Entities on Enter

When a `TodoText` component is added to an entity, one system gives it a DOM element, builds child entities, and registers event handlers on the todo entity itself.

<div data-code-tabs>

```tsx
import { defineReactiveSystem, Entities, defineEvent, defineMarker, materialize } from "@ecs-test/ecs";

const TodoToggleButton = defineMarker("TodoToggleButton");
const TodoLabel = defineMarker("TodoLabel");
const TodoToggleRequested = defineEvent("todo.toggleRequested");
const TodoRemoveRequested = defineEvent("todo.removeRequested");

function buildTodoRow(world, todoEntity) {
  const text = world.get(todoEntity, TodoText)?.value ?? "";

  // Toggle button emits an intent event.
  materialize(world, [
    <Entity>
      <DOMElement tag="button" />
      <TodoToggleButton />
      <TextContent value="\u2610" />
      <Style border="none" background="none" fontSize="18px"
             cursor="pointer" padding="0" />
      <Clickable onClick={(world, entity) => {
        world.emit(entity, TodoToggleRequested);
      }} />
    </Entity>,

    // Label is updated by updateTodoVisuals().
    <Entity>
      <DOMElement tag="span" />
      <TodoLabel />
      <TextContent value={text} />
      <Style flex="1" textDecoration="none" opacity="1" />
    </Entity>,
    <Entity>
      <DOMElement tag="button" />
      <TextContent value="Remove" />
      <Style border="none" background="none" color="#ef4444"
             fontSize="18px" cursor="pointer" padding="0 4px" />
      <Clickable onClick={(world, entity) => {
        world.emit(entity, TodoRemoveRequested);
      }} />
    </Entity>,
  ], todoEntity);
}

function updateTodoVisuals(world, todoEntity) {
  const done = world.get(todoEntity, Status)?.completed ?? false;

  const toggle = world.findChildMaybe(todoEntity, TodoToggleButton);
  if (toggle != null) {
    world.set(toggle, TextContent({ value: done ? "\u2611" : "\u2610" }));
  }

  const label = world.findChildMaybe(todoEntity, TodoLabel);
  if (label != null) {
    world.set(label, Style({
      flex: "1",
      textDecoration: done ? "line-through" : "none",
      opacity: done ? "0.5" : "1",
    }));
  }
}

const TodoEnterSystem = defineReactiveSystem({
  name: "TodoEnterSystem",
  query: Entities.with([TodoText]),
  onEnter(world, entities) {
    for (const entity of entities) {
      // Row shell
      world.add(entity, DOMElement({ tag: "div" }));
      world.add(entity, Style({
        display: "flex",
        alignItems: "center",
        gap: "8px",
        padding: "6px 0",
        borderBottom: "1px solid #e5e7eb",
      }));

      // Row children + initial visuals
      buildTodoRow(world, entity);
      updateTodoVisuals(world, entity);

      // Root-level intent handlers for this todo row
      world.on(entity, TodoToggleRequested, () => {
        const status = world.get(entity, Status) ?? { completed: false };
        world.set(entity, Status({ completed: !status.completed }));
        updateTodoVisuals(world, entity);
      });

      world.on(entity, TodoRemoveRequested, () => {
        world.removeEntity(entity);
      });
    }
  },
});
```

```ts
const TodoToggleButton = defineMarker("TodoToggleButton");
const TodoLabel = defineMarker("TodoLabel");
const TodoToggleRequested = defineEvent("todo.toggleRequested");
const TodoRemoveRequested = defineEvent("todo.removeRequested");

function buildTodoRow(world, todoEntity) {
  const text = world.get(todoEntity, TodoText)?.value ?? "";

  // Toggle button emits an intent event.
  world.createEntity(todoEntity, [
    DOMElement({ tag: "button" }),
    TodoToggleButton(),
    TextContent({ value: "\u2610" }),
    Style({ border: "none", background: "none", fontSize: "18px", cursor: "pointer", padding: "0" }),
    Clickable({ onClick: (world, entity) => {
      world.emit(entity, TodoToggleRequested);
    }}),
  ]);

  // Label is updated by updateTodoVisuals().
  world.createEntity(todoEntity, [
    DOMElement({ tag: "span" }),
    TodoLabel(),
    TextContent({ value: text }),
    Style({ flex: "1", textDecoration: "none", opacity: "1" }),
  ]);
  world.createEntity(todoEntity, [
    DOMElement({ tag: "button" }),
    TextContent({ value: "Remove" }),
    Style({ border: "none", background: "none", color: "#ef4444", fontSize: "18px", cursor: "pointer", padding: "0 4px" }),
    Clickable({ onClick: (world, entity) => {
      world.emit(entity, TodoRemoveRequested);
    }}),
  ]);
}

function updateTodoVisuals(world, todoEntity) {
  const done = world.get(todoEntity, Status)?.completed ?? false;

  const toggle = world.findChildMaybe(todoEntity, TodoToggleButton);
  if (toggle != null) {
    world.set(toggle, TextContent({ value: done ? "\u2611" : "\u2610" }));
  }

  const label = world.findChildMaybe(todoEntity, TodoLabel);
  if (label != null) {
    world.set(label, Style({
      flex: "1",
      textDecoration: done ? "line-through" : "none",
      opacity: done ? "0.5" : "1",
    }));
  }
}

const TodoEnterSystem = defineReactiveSystem({
  name: "TodoEnterSystem",
  query: Entities.with([TodoText]),
  onEnter(world, entities) {
    for (const entity of entities) {
      // Row shell
      world.add(entity, DOMElement({ tag: "div" }));
      world.add(entity, Style({
        display: "flex",
        alignItems: "center",
        gap: "8px",
        padding: "6px 0",

      }));

      buildTodoRow(world, entity);
      updateTodoVisuals(world, entity);

      // Root-level intent handlers for this todo row
      world.on(entity, TodoToggleRequested, () => {

        world.set(entity, Status({ completed: !status.completed }));

      });

      world.on(entity, TodoRemoveRequested, () => {
        world.removeEntity(entity);
      });
    }
  },

```

</div>

The `onClick` handler on `Clickable` now emits typed events (`todo.toggleRequested`, `todo.removeRequested`), and handlers are registered on the todo entity itself. That means no repeated `getParent()` plumbing in button handlers.

`findChildMaybe(todo, TodoToggleButton/TodoLabel)` enables targeted updates to the exact child entities that need visual changes without `[0]` indexing.

## 4. Update Visuals on State Change

When `Status.completed` changes, visuals are updated directly by the todo entity's toggle handler:

```ts
world.on(entity, TodoToggleRequested, () => {
  const status = world.get(entity, Status) ?? { completed: false };
  world.set(entity, Status({ completed: !status.completed }));
  updateTodoVisuals(world, entity);
});
```

No child teardown/rebuild is needed: `findChildMaybe(...)` (or `findChild(...)` when required) finds exact row children to patch.

## 5. Handle Adding Todos

The Add button's `onClick` handler (from step 2) calls `addTodoFromInput` — here is that helper in full:

```ts
import { getDOMElement } from "@ecs-test/dom";

function addTodoFromInput(world) {
  // No built-in input component — use getDOMElement to access the raw DOM node
  const inputEntity = world.query(TodoInput)[0];
  if (inputEntity == null) return;
  const inputEl = getDOMElement(world, inputEntity) as HTMLInputElement;
  if (!inputEl) return;

  const text = inputEl.value.trim();
  if (!text) return;
  inputEl.value = "";

  // Create the todo as a child of the list container.
  // TodoEnterSystem will pick it up and build its DOM children.
  const listEntity = world.query(TodoListContainer)[0];
  if (listEntity == null) return;
  world.createEntity(listEntity, [TodoText({ value: text }), Status({ completed: false })]);
}
```

Notice there's no system for this — `onClick` has full access to the world, so it can query for the input and list container directly. The handler is defined as a helper function so it can be reused (e.g., for Enter key handling).

## 6. Wire and Run

<div data-code-tabs>

```tsx
import { World } from "@ecs-test/ecs";
import { registerDOMSystems } from "@ecs-test/dom";

const world = new World({
  externals: {
    createElement: document.createElement.bind(document),
    rootContainer: document.getElementById("root")!,
    window,
    console,
  },
});

registerDOMSystems(world);
world.registerSystem(TodoEnterSystem);

// materialize(world, ui) — the shell from step 2

// Seed initial todos under the list container.
// Each is an entity with TodoText and Status.
const listEntity = world.query(TodoListContainer)[0];

world.createEntity(listEntity, [
  TodoText({ value: "Learn about ECS entities" }),
  Status({ completed: true }),  // This one starts completed
]);
world.createEntity(listEntity, [
  TodoText({ value: "Build a todo app" }),
  Status({ completed: false }),
]);

world.flush();
```
